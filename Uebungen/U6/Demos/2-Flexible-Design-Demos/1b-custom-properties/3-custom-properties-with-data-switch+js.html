CSS Custom Properties / CSS Variables in Action


  CSS
    :root {
      /*colors*/
      --color-light-red: #ffeaea;
      --color-dark-violet: #47027e;
    }

    /*variables / switching*/
    body[data-style='light'] {
      --color: var(--color-dark-violet);
      --bg-color: var(--color-light-red);
    }

    /*variables / switching*/
    body[data-style='dark'] {
      --color: var(--color-light-red);
      --bg-color: var(--color-dark-violet);
    }

    /*applying variables*/
    body {
      color: var(--color);
      background-color: var(--bg-color);
    }

  JS
  document.querySelector('fieldset').addEventListener('change', (event) => {
    const style = event.target.value;
    document.body.setAttribute('data-style', style);
  })
