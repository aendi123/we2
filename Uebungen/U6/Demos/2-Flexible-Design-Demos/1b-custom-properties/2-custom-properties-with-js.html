CSS Custom Properties / CSS Variables in Action


  CSS
  :root {
      /*colors*/
      --color-light-red: #ffeaea;
      --color-dark-violet: #47027e;

      /*variables*/
      --color: var(--color-dark-violet);
      --bg-color: var(--color-light-red);
    }

    /*applying variables*/
    body {
      color: var(--color);
      background-color: var(--bg-color);
    }

  JS
  document.querySelector('fieldset').addEventListener('change', (event) => {
    const style = event.target.value;

    const styles = getComputedStyle(document.documentElement);
    const lightRedColor = String(styles.getPropertyValue('--color-light-red')).trim();
    const darkVioletColor = String(styles.getPropertyValue('--color-dark-violet')).trim();

    if (style === 'light'){
      document.documentElement.style.setProperty('--color', darkVioletColor);
      document.documentElement.style.setProperty('--bg-color', lightRedColor);
    }else{
      document.documentElement.style.setProperty('--color', lightRedColor);
      document.documentElement.style.setProperty('--bg-color', darkVioletColor);
    }
  }
